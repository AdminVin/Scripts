#> Identify SD Card
lsblk

Example Output:
sda 8:0 1 29.2G 0 disk 
├─sda1 8:1 1 512M 0 part /media/user/bootfs 
└─sda2 8:2 1 28.6G 0 part /media/user/rootfs

#> Image - Capture
sudo dd if=/dev/sda of=~/Desktop/raspi-backup.img bs=4M status=progress conv=fsync

#> Image - Remove Card
sync

#> Image - Shrink

# Install PiShrink
sudo apt update
sudo apt install -y git parted e2fsprogs
git clone https://github.com/Drewsif/PiShrink.git
cd PiShrink
sudo cp pishrink.sh /usr/local/bin/pishrink
sudo chmod +x /usr/local/bin/pishrink

# Shrink Image
sudo pishrink ~/Desktop/raspi-backup.img

# Image Verification
echo $?

Expected Output:
adminvin@VinLX01:~/PiShrink$ echo $? 0

0 infers no errors.
