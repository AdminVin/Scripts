# Install & Configure

#> Windows: Can be run from a CMD/Terminal Prompt
#> Synology: Run as scheduled Task (as root) / SSH > "sudo -i" (without quotes)

# Install with a daily schedule of 4 AM
docker run -d \
  --name=Kometa \
  -e TZ=America/New_York \
  -e PMM_TIME=04:00 \
  -v /volume1/docker/Kometa:/config \
  --network host \
  --restart unless-stopped \
  kometateam/kometa:latest

# Run - Schedule (Instant)
docker exec -it Kometa python3 -m kometa --config /config/config.yml --run --debug
