# Install & Configure

#> Windows: Can be run from a CMD/Terminal Prompt
#> Synology: Run as scheduled Task (as root) / SSH > "sudo -i" (without quotes)

# Docker Image: https://hub.docker.com/r/sctx/overseerr

docker run -d \
  --name=overseerr \
  -e LOG_LEVEL=debug \
  -e TZ=US/Eastern \
  -e PORT=5055 \
  -p 5055:5055 \
  -v /volume1/docker/Overseerr:/app/config \
  --restart unless-stopped \
  sctx/overseerr

# Update Version
docker stop Overseerr
docker rm Overseerr
docker pull sctx/overseerr
